```

МассивСтрок = СтрРазделить(Строка," ",Ложь);

Если МассивСтрок.Количество() >= 3 Тогда 
	
	ДеньЦифрами = МассивСтрок[0];
	Если СтрДлина(ДеньЦифрами) = 1 Тогда
		ДеньЦифрами = "0" + ДеньЦифрами;
	КонецЕсли;

		Если ВРег(МесяцБуквами) = "ЯНВАРЯ" Тогда
			МесяцЦифрами = "01";

			МесяцЦифрами = "02";
		ИначеЕсли ВРег(МесяцБуквами) = "МАРТА" Тогда
			МесяцЦифрами = "03";
		ИначеЕсли ВРег(МесяцБуквами) = "АПРЕЛЯ" Тогда
			МесяцЦифрами = "04";
		ИначеЕсли ВРег(МесяцБуквами) = "МАЯ" Тогда
			МесяцЦифрами = "05";
		ИначеЕсли ВРег(МесяцБуквами) = "ИЮНЯ" Тогда
			МесяцЦифрами = "06";
		ИначеЕсли ВРег(МесяцБуквами) = "ИЮЛЯ" Тогда
			МесяцЦифрами = "07";
		ИначеЕсли ВРег(МесяцБуквами) = "АВГУСТА" Тогда
			МесяцЦифрами = "08";
		ИначеЕсли ВРег(МесяцБуквами) = "СЕНТЯБРЯ" Тогда
			МесяцЦифрами = "09";
		ИначеЕсли ВРег(МесяцБуквами) = "ОКТЯБРЯ" Тогда
			МесяцЦифрами = "10";
		ИначеЕсли ВРег(МесяцБуквами) = "НОЯБРЯ" Тогда
			МесяцЦифрами = "11";

			МесяцЦифрами = Неопределено;
		КонецЕсли;
		
	
	ГодЦифрами = МассивСтрок[2];
	СтрокаСДатой = ГодЦифрами + МесяцЦифрами + ДеньЦифрами;	
	Результат = ОписаниеТипДата.ПривестиЗначение(СтрокаСДатой);
	
	Возврат Результат;
	
	Иначе
	Результат = Неопределено;
КонецЕсли;
```
